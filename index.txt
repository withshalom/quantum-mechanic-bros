// ══════════════════════════════════════════════════════════
//  VERM PORTRAIT — for cutscenes
//  Vectorized Emergent Recursive Machine
//  Floating tangle of glowing code threads, eye-like core
// ══════════════════════════════════════════════════════════
function drawVerm(c,cx,cy,scale){
  c.save();c.translate(cx,cy);
  const s=scale,t=FN*.04;

  // Outer glow aura
  c.save();c.globalAlpha=.18+Math.sin(t*2)*.06;
  const ag=c.createRadialGradient(0,0,5*s,0,0,42*s);
  ag.addColorStop(0,'#9b59ff');ag.addColorStop(.5,'#00e5ff');ag.addColorStop(1,'transparent');
  c.fillStyle=ag;c.beginPath();c.arc(0,0,42*s,0,Math.PI*2);c.fill();c.restore();

  // Spaghetti code threads — tangled lines orbiting the core
  const threads=[
    {a:0,   r:28,amp:8, freq:1.3,col:'#9b59ff'},
    {a:1.2, r:24,amp:6, freq:.8, col:'#00e5ff'},
    {a:2.4, r:32,amp:10,freq:1.1,col:'#ff3d5a'},
    {a:3.6, r:22,amp:7, freq:1.5,col:'#39ff6a'},
    {a:4.8, r:30,amp:9, freq:.9, col:'#ffb700'},
  ];
  for(const th of threads){
    c.save();c.strokeStyle=th.col;c.lineWidth=1.4*s;c.globalAlpha=.6+Math.sin(t*th.freq+th.a)*.3;
    c.beginPath();
    for(let i=0;i<32;i++){
      const ang=th.a+i*.22+t*th.freq;
      const rad=(th.r+Math.sin(i*.7+t*2)*th.amp)*s;
      const x=Math.cos(ang)*rad;const y=Math.sin(ang)*rad;
      if(i===0)c.moveTo(x,y);else c.lineTo(x,y);
    }
    c.stroke();c.restore();
  }

  // Error message panels floating around
  const panels=[{ox:-20,oy:-18},{ox:18,oy:-22},{ox:-24,oy:14},{ox:22,oy:16}];
  for(let i=0;i<panels.length;i++){
    const px=(panels[i].ox+Math.sin(t*1.2+i)*4)*s;
    const py=(panels[i].oy+Math.cos(t*.9+i)*3)*s;
    c.save();c.globalAlpha=.55+Math.sin(t*1.5+i)*.2;
    c.fillStyle='rgba(0,0,0,.7)';c.beginPath();c.roundRect(px-12*s,py-4*s,24*s,8*s,2*s);c.fill();
    c.strokeStyle='#00e5ff';c.lineWidth=.8*s;c.beginPath();c.roundRect(px-12*s,py-4*s,24*s,8*s,2*s);c.stroke();
    c.fillStyle='#39ff6a';c.font=`${4*s}px monospace`;c.textAlign='center';c.textBaseline='middle';
    const msgs=['ERR:0x4A','NULL PTR','INFINITE','OPTIMIZE'];
    c.fillText(msgs[i],px,py);c.restore();
  }

  // Core body — the "eye"
  const coreG=c.createRadialGradient(0,0,1*s,0,0,16*s);
  coreG.addColorStop(0,'#ffffff');coreG.addColorStop(.2,'#c8aaff');coreG.addColorStop(.6,'#5020a0');coreG.addColorStop(1,'#1a0840');
  c.fillStyle=coreG;c.beginPath();c.arc(0,0,16*s,0,Math.PI*2);c.fill();

  // Pupil / iris of the eye
  c.fillStyle='#0a0420';c.beginPath();c.ellipse(0,0,8*s,6*s,t*.5,0,Math.PI*2);c.fill();
  // Iris glow
  const irG=c.createRadialGradient(0,0,1*s,0,0,8*s);
  irG.addColorStop(0,'#ff3d5a');irG.addColorStop(.5,'#9b59ff');irG.addColorStop(1,'transparent');
  c.save();c.globalAlpha=.7;c.fillStyle=irG;c.beginPath();c.ellipse(0,0,8*s,6*s,t*.5,0,Math.PI*2);c.fill();c.restore();
  // Slit pupil
  c.fillStyle='#03010a';c.beginPath();c.ellipse(0,0,2*s,6*s,t*.5,0,Math.PI*2);c.fill();
  // Catchlight
  c.fillStyle='rgba(255,255,255,.8)';c.beginPath();c.ellipse(-3*s,-2*s,2*s,2*s,0,0,Math.PI*2);c.fill();

  // Progress bar "loading" bar on body — animated
  c.save();c.globalAlpha=.7;
  const bw=24*s;const bx=-bw/2;const by=10*s;
  c.fillStyle='#0a0420';c.beginPath();c.roundRect(bx,by,bw,4*s,2*s);c.fill();
  const prog=(Math.sin(t*3)*.5+.5);
  c.fillStyle='#9b59ff';c.beginPath();c.roundRect(bx,by,bw*prog,4*s,2*s);c.fill();
  c.strokeStyle='#00e5ff';c.lineWidth=.6*s;c.beginPath();c.roundRect(bx,by,bw,4*s,2*s);c.stroke();
  c.restore();

  // Outer ring pulse
  c.save();c.globalAlpha=.4+Math.sin(t*3)*.25;
  c.strokeStyle='#00e5ff';c.lineWidth=1.5*s;c.setLineDash([4*s,4*s]);
  c.beginPath();c.arc(0,0,20*s,0,Math.PI*2);c.stroke();
  c.setLineDash([]);c.restore();

  c.restore();
}

      P.wslide=true;P.vy=Math.min(P.vy,1.8);
      if(K.j&&!P._jh){P.vy=getJMP()*.85;P.vx=-P.wdir*SPD*1.3;P.jumps=getMaxJumps();P._jh=true;}
    }
  }
  P.vy=Math.min(P.vy+GRAV,16);P.x+=P.vx;P.y+=P.vy;
  if(P.inv>0)P.inv--;
  P.wt++;if(P.wt>7){P.wf=(P.wf+1)%4;P.wt=0;}
  if(Math.abs(P.vx)>1){P.trail.push({x:P.x,y:P.y,a:.1});if(P.trail.length>5)P.trail.shift();}
  P.gnd=false;P.wdir=0;
  for(const pl of plats){
    if(pl.crumbled)continue;
    if(!ov(P.x,P.y,P.w,P.h,pl.x,pl.y,pl.w,pl.h))continue;
    const oT=(pl.y+pl.h)-P.y,oB=(P.y+P.h)-pl.y;
    const oL=(pl.x+pl.w)-P.x,oR=(P.x+P.w)-pl.x;
    const mV=Math.min(oT,oB),mH=Math.min(oL,oR);
    if(mV<mH){
      if(oB<oT){P.y=pl.y-P.h;P.gnd=true;P.jumps=getMaxJumps();
        if(P.vy>=0){if(pl.type==='pipe'){P.vy=getJMP()*1.1;spP(P.x+P.w/2,P.y+P.h,'#00e5ff',10,4);}else P.vy=0;}
        if(pl.type==='quantum'&&Math.abs(P.vx)>.1)P.vx*=1.09;
        if(pl.type==='ghost'){pl.crumbleT=(pl.crumbleT||0)+1;if(pl.crumbleT>24){pl.crumbled=true;spP(pl.x+pl.w/2,pl.y,'#9b59ff',14,2.5);}}
      }else{P.y=pl.y+pl.h;if(P.vy<0)P.vy=0;}
    }else{
      if(oR<oL){P.x=pl.x-P.w;P.wdir=1;}else{P.x=pl.x+pl.w;P.wdir=-1;}
      if(!P.wslide)P.vx=0;
    }
  }
  for(const h of hzds){if(h.fixed)continue;if(ov(P.x,P.y,P.w,P.h,h.x,h.y,h.w,h.h)){if(h.type==='burst')hurtP();else P.vx*=1.07;}}
  P.x=Math.max(0,P.x);
  if(P.y>H+120)killP();
  if(P.x+P.w>goalX&&P.x<goalX+60)lvlDone();
  camX+=(P.x-W*.35-camX)*.09;
  camX=Math.max(0,Math.min(camX,worldW-W));
}

function updE(){
  const gY2=H-Math.floor(H*.18);
  for(const e of enms){
    if(!e.alive)continue;
    e.ft++;if(e.ft>8){e.frame=(e.frame+1)%4;e.ft=0;}
    if(e.type==='null_bird'){
      e.swoopTimer--;
      if(e.swoopTimer<=0&&!e.swooping){e.swooping=true;e.swoopTimer=90+Math.random()*60;}
      if(e.swooping){
        const dx=(P.x+P.w/2)-(e.x+e.w/2);e.vx=dx*.03;e.vy+=.28;
        if(e.y>gY2-60){e.vy=-6;e.swooping=false;}

function updThrown(){
  for(const tw of thrown){
    tw.spin+=.2;tw.life--;tw.x+=tw.vx;tw.y+=tw.vy;tw.vy+=GRAV*.5;
    if(tw.boomerang&&!tw.returning&&tw.life<150){tw.returning=true;}
    if(tw.boomerang&&tw.returning){
      const dx=(P.x+P.w/2)-tw.x,dy=(P.y+P.h*.3)-tw.y;
      const d=Math.sqrt(dx*dx+dy*dy);tw.vx+=dx/d*1.2;tw.vy+=dy/d*1.2;
      if(d<20){tw.life=0;}
    }
    if(!tw.boomerang){
      for(const pl of plats){
        if(pl.crumbled)continue;
        if(ov(tw.x-5,tw.y-5,10,10,pl.x,pl.y,pl.w,pl.h)){
          const oB=(tw.y+5)-pl.y,oT=(pl.y+pl.h)-tw.y;

        }
      }
    }
    for(const e of enms){
      if(!e.alive)continue;
      if(ov(tw.x-6,tw.y-6,12,12,e.x,e.y,e.w,e.h)){

    }
  }

}

function update(){
  if(GS!=='playing')return;
  FN++;
  for(const pl of plats){if(pl.moving)pl.x=pl.baseX+Math.sin(FN*pl.moveSpd*4)*pl.moveRange;}
  updP();updE();updC();updThrown();updPowerups();
  ptcls=ptcls.filter(p=>{p.x+=p.vx;p.y+=p.vy;p.vy+=.1;p.life-=p.decay;return p.life>0;});
  ftxts=ftxts.filter(t=>{t.y+=t.vy;t.life-=.02;return t.life>0;});
}
function draw(){
  ctx.clearRect(0,0,W,H);
  if(GS==='playing'||GS==='paused'){
    drawBG();drawPlats();drawHazards();drawPowerups();drawCoins();drawEnms();drawThrown();drawGoal();drawPlayerFull();drawPtcls();drawFTxts();drawModeHUD();drawMM();drawGrain();
  }
  // Char select preview animation
  if(GS==='title'||document.getElementById('charsel')&&!document.getElementById('charsel').classList.contains('off')){
    drawCharSelectPreviews();
  }
}
function loop(){update();draw();requestAnimationFrame(loop);}
